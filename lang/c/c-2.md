## compile flow

```text
                gcc     <- .c, .h (source)
                |  cpp
                |       -> .i  (preprocess source)
                |  cc1
                |       -> .s  (assembly)
                |  as
                |       -> .o  (relocatable)
                |  collect2
                |
                v
library     |
c runt-ime  |  --- ld   -> bin (executable)
link script |
```

## preprocessor / macro

### define

```c
#include <stdio.h>
#include <string.h>

// object-like macro
#define PI 3.14

// function-like macro
#define MIN(a,b) ((a)<(b)?(a):(b))

// stringize
#define xstr(s)  #s

// concatenate
#define concat(a, b) a##b

// variadic macro
#define pr(s, ...) printf(s, __VA_ARGS__)
// #define pr(s, args...) printf(s, args)

int main( ) {
   float r = 10.0;
   int xy = 30;
   printf("circle area: %f\n", PI * r * r);
   printf("min(1, -1): %d\n", MIN(1, -1));
   printf("xstr(4): %s\n", xstr(4));
   printf("xy = %d\n", concat(x, y));
   pr("%s\n", "hello");

   // predefined macro
   printf("File :%s\n", __FILE__ );
   printf("Date :%s\n", __DATE__ );
   printf("Time :%s\n", __TIME__ );
   printf("Line :%d\n", __LINE__ );
   printf("STDC :%d\n", __STDC__ );

   return 0;
}
```

### trick or trip

```c

   printf("DOUBLE(1+1): %d\n", DOUBLE(1 + 1));
```

```c

int main()
{
   int x = 1, y = 2;

   int m = max(x, y++);
   printf("m = max(x, y++) = %d, y = %d\n", m, y);

// m = max(x, y++) = 3, y = 4

   return 0;
}
```

```c
#include <stdio.h>

// multi line marco 需要加 {}
#define increse(a, b) \
    (a)++;            \
    (b)++;

#define expand(a, b) \
   {                \
      (a)++;       \
      (b)++;       \
   }

int main()
{
   int a, b;

   a = 1, b = 2;
   printf("before a: %d, b: %d\n", a, b);
   increse(a, b);
   printf("after a: %d, b: %d\n", a, b);
// before a: 1, b: 2
// after a: 2, b: 3

   a = 1, b = 2;
   printf("before a: %d, b: %d\n", a, b);
   if (a < b)
      increse(a, b);
   printf("after a: %d, b: %d\n", a, b);
// before a: 1, b: 2
// after a: 2, b: 3

   a = 1, b = 2;
   printf("before a: %d, b: %d\n", a, b);
   if (a > b)
      increse(a, b);
   printf("after a: %d, b: %d\n", a, b);
// before a: 1, b: 2
// after a: 1, b: 3

   a = 1, b = 2;
   printf("before a: %d, b: %d\n", a, b);
   if (a > b)
      expand(a, b);
   printf("after a: %d, b: %d\n", a, b);
// before a: 1, b: 2
// after a: 1, b: 2

   return 0;
}
```

## memory

### allocate

```c
#include <stdio.h>
#include <stdlib.h>
int main()
{
   int *ptr, i , n1, n2;

   printf("Enter size: ");
   scanf("%d", &n1);

   // ptr = (int*) calloc(n1, sizeof(int));
   ptr = (int*) malloc(n1 * sizeof(int));
   printf("Addresses of previously allocated memory: ");
   for(i = 0; i < n1; ++i)
      printf("%u\n",ptr + i);

   printf("\nEnter the new size: ");
   scanf("%d", &n2);

   ptr = realloc(ptr, n2 * sizeof(int));
   printf("Addresses of newly allocated memory: ");
   for(i = 0; i < n2; ++i)
      printf("%u\n", ptr + i);

   free(ptr);

   return 0;
}
```

### bit operator

```c
#include <stdio.h>

int main()
{
    char p;
    char q;
    printf("%5s%5s%5s%5s%5s%5s%5s\n", "p", "q", "p&q", "p|q", "p^q", "!p", "~p");
    for (p = 0; p < 2; p++)
    {
        for (q = 0; q < 2; q++)
        {
            printf("%5d%5d%5d%5d%5d%5d%5d\n", p, q, p & q, p | q, p ^ q, !p, ~p);
        }
    }

    printf("logical shift\tarithmetic shift\n");
    char v = 5;
    for (int i = 0; i < 5; i++)
    {
        printf("%d >> %d = %d\t", v, i, v >> i);
        printf("%d >> %d = %d\n", -v, i, -v >> i);
    }
}
```

```text
p&q / and

p|q / or

p^q / xor

   p = q^(p^q) = q^(q^p)
     = (p^q)^q = (q^p)^q

!p  / not

~p  / 1's complement

    1        = 0000 0001
   ~1        = 1111 1110

~p + 1 / 2's complement

    1        = 0000 0001
   ~1        = 1111 1110  =  -2
   ~1 + 1    = 1111 1111  =  -1

   -1        = 1111 1111
   ~(-1)     = 0000 0000  =  0
   ~(-1) + 1 = 0000 0001  =  1


<< / left shift

    1        = 0000 0001
    1 << 3   = 0000 1000
                     ^^^
                     |||
                     fill 0

>> / right shift

   45        = 0010 1101
   45 >> 3   = 0000 0101  =  5
               ^^^
               |||
               fill 0

logical shift

   when right shift, fill 0

   -45       = 1101 0011
   -45 >> 3  = 0001 1010  =  26
               ^^^
               |||
               fill 0

arithmetic shift / signed shift

   when right shift, fill by sign bit

   -45       = 1101 0011
   -45 >> 3  = 1111 1010  =  -6
               ^^^
               |||
               fill 1
```

```c
#include <stdio.h>

int main()
{
   int flag = 5;

   (flag & 0) ? printf("& 0 t\n") : printf("& 0 f\n"); // & 0 f, always false
   (flag & 1) ? printf("& 1 t\n") : printf("& 1 f\n"); // & 1 t
   (flag & 3) ? printf("& 3 t\n") : printf("& 3 f\n"); // & 3 t

   unsigned int a = 60; /* 60 = 0011 1100 */
   unsigned int b = 13; /* 13 = 0000 1101 */

   // and, &
   printf("a & b = %d\n", a & b); // 12 = 0000 1100
   printf("a & a = %d\n", a & a); // 60 = 0011 1100

   // or, |
   printf("a | b = %d\n", a | b); // 61 = 0011 1101
   printf("a | a = %d\n", a | a); // 60 = 0011 1100

   // xor, ^
   printf("a ^ b = %d\n", a ^ b);        // 49 = 0011 0001
   printf("a ^ a = %d\n", a ^ a);        // 0  = 0000 0000
   printf("a ^ a ^ a= %d\n", a ^ a ^ a); // 60 = 0011 1100

   printf("~a = %d\n", ~a);         // -61 = 1100 0011
   printf("a << 2 = %d\n", a << 2); // 240 = 1111 0000
   printf("a >> 2 = %d\n", a >> 2); //  15 = 0000 1111

   return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

char *char2Bin(unsigned char value)
{
   char *c = malloc(sizeof(char) * 8);
   int a;
   for (int i = sizeof(char) * 7; i >= 0; i--)
   {
      a = (value & (1 << i)) >> i;
      *(c + i) = a + '0'; // int to char
   }
   return c;
}

int main()
{
   char *a = char2Bin('a');
   char *A = char2Bin('A');
   printf("a: %s\nA: %s\n", a, A);

   char c = 'A';
   printf("to lower : %c -> %c\n", c, c | ' ');
   printf("to upper : %c -> %c\n", c, c & '_');
   printf("to switch: %c -> %c\n", c, c ^ ' ');

   int x = 2147483646;
   int y = 2147483644;
   int z = (x&y)+((x^y) >> 1);
   printf("(%d + %d ) / 2 = %d\n", x, y, z);

   return 0;
}
```
