arp
arp -a [host] 				ip n show
arp -d [ip_addr] 			ip n del [ip_addr], ip neighbor flush [ip_addr]
arp -i [int] 				ip n [add | chg | del | repl] dev [name]
arp -s [ip_addr] [hw_addr]	ip n add [ip_addr] lladdr [mac_address] dev [device] nud [nud_state]
